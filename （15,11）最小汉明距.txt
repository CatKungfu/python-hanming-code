import np

def hm_result(hm_code):
    # 列表a用来装汉明码距离
    x = []
    #print(len(hm_code))
    for i in range(len(hm_code)):
        xmat = np.mat(hm_code)
        for j in range(i+1,len(hm_code)):
            if i < len(hm_code)-1:
                smstr = np.nonzero(xmat[i] - xmat[i+1])
                x.append(np.shape(smstr[1])[0])
    #print(x)
    return min(x)
